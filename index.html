Code Sample – Simple Margin Calculator (Python)


import math

def margin(revenue, cost):
    if revenue == 0:
        return 0.0
    return (revenue - cost) / revenue

r = 2450.0
c = 1815.0
print(f"Operating margin: {margin(r, c):.2%}")
